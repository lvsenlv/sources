1. 第一行

		#!/bin/bash
"#!" 是一个约定的标记，它告诉系统这个脚本需要什么解释器来执行，即使用哪一种Shell

2. if 语句
  
		#结构：
		if [ ]
		then
		  ......
		else
		  ......
		fi
**注意：在 if 和 [ ] 之间必须有一个空格**

3. case 语句  

		#结构
		case $变量名 in
		情况1 )
		    ......
		    ;;
		情况2 )
		    ......
		    ;;
		* )       #相当于C语言中的default语句
		    ......
		    ;;
		esac
